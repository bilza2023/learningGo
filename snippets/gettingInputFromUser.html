Getting Input from the user

We have to send var to Scan by reference hence &userName.


package main


import "fmt"

func main(){
    var name string = "John"
    var age uint = 22
    fmt.Printf("My name is %v and my age is :%v \n",name,age);
    fmt.Println("What is your name please?") 
    var userName string
    //--Note the & sign to sned the var by reference
    fmt.Scan(&userName)
    fmt.Println("Welcome..", userName) 
}
