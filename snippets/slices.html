Arrays

Array basic example.


package main
import "fmt"

func main(){
// - Also
// var cities = [] string { "London" , "Milan" , "Tokyo"}
// cities := [] string { "London" , "Milan" , "Tokyo"}

var cities = [] string {}
cities = append(cities,"London" )
cities = append(cities,"Milan" )
cities = append(cities,"Tokyo" )

//--Slice Type and value
	fmt.Printf("Slice value : %v \n" , cities)
	fmt.Printf("Slice Type : %T \n" , cities)
//--Slice Length using len()	
	fmt.Printf("Slice length : %v \n" , len(cities) )

////--Alternative way
names := []string {}
names = append(names,"Tim" )
names = append(names,"Ali" )
fmt.Println("Slice :" , names);

//--Slice also has Capacity

fmt.Println("The capacity of the slice is" , cap(names));

}
