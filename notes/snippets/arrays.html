Arrays

Array basic example.


package main
import "fmt"

func main(){

var users = [3]string { "mike", "jim" , "mich"}
fmt.Println("Welcome..", users[0]) 
fmt.Println("Welcome..", users[1]) 
fmt.Println("Welcome..", users[2]) 

var cities = [3] string {}
// also cities := [3] string {}
	cities[0] = "London"
	cities[1] = "Milan"
	cities[2] = "Tokyo"
	//--This line will cause error
	// cities[10] = "Error"

fmt.Println( "The names of the cities are :", 
	cities[0] , "," ,
	cities[1],  ",",
	cities[2])

//--Array Type and value
	fmt.Printf("Array value : %v \n" , cities)
	fmt.Printf("Array Type : %T \n" , cities)
//--Array Length using len()	
	fmt.Printf("Array length : %v \n" , len(cities) )
}
